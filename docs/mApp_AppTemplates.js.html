mApp/AppTemplates.js

import { DefaultMethods } from "./DefaultMethods";
/**
 * Type of object containing app templates
 * @typedef {Object} AppTemplates
 * @property {Function} template_default - The default starting template. All methods come from this.DefaultMethods.
 * @ignore
 */

/** @namespace MerjentApp.AppTemplates */

/** 
 * @memberof MerjentApp.AppTemplates
 * @returns {AppTemplates}
 * @param {MerjentApp} appSettings
 */
function AppTemplates(appSettings){
    var methods = DefaultMethods(appSettings)
    return {
        /**
         * @description Default app template. Binds default functions to MerjentApp.
         * 
         * @memberof MerjentApp.AppTemplates#
         * @method
         * @param {*} event
         * @returns {void}
         */
        template_default(){
            appSettings.getPermissions=methods.getPermissions.bind(appSettings);
            appSettings.getStatus=methods.getStatus.bind(appSettings);
            appSettings.getCompleteLockStatus=methods.getCompleteLockStatus.bind(appSettings);
            appSettings.baseUpdatePermission=methods.baseUpdatePermission.bind(appSettings);
            appSettings.addUpdatePermission=methods.addUpdatePermission.bind(appSettings);
            appSettings.updatePermission=methods.updatePermission.bind(appSettings);
            appSettings.updateStatus=methods.updateStatus.bind(appSettings);
            appSettings.getLockStatus=methods.getLockStatus.bind(appSettings);
            appSettings.updateLock=methods.updateLock.bind(appSettings);
            appSettings.baseLockEditing = methods.baseLockEditing.bind(appSettings)
            appSettings.addLockEditing = methods.addLockEditing.bind(appSettings)
            appSettings.lockEditing=methods.lockEditing.bind(appSettings);
            appSettings.validatePhoto=methods.validatePhoto.bind(appSettings);
            appSettings.start=methods.start.bind(appSettings);
            appSettings.reportIDprefix=methods.reportIDprefix.bind(appSettings)
            appSettings.enterReportID=methods.enterReportID.bind(appSettings);
            appSettings.addCoordinatorInitials = methods.addCoordinatorInitials.bind(appSettings)
            appSettings.limitCaption = methods.limitCaption.bind(appSettings)
            appSettings.validateAllPhotos = methods.validateAllPhotos.bind(appSettings)
            appSettings.disableNewRecordCreation = methods.disableNewRecordCreation.bind(appSettings)
        }
    }

}

export { AppTemplates }
