import { StatusMixins } from '../mixins/StatusMixins';
import { PermissionMixins } from '../mixins/PermissionMixins';
import { LockingMixins } from '../mixins/LockingMixins';
import { PhotoMixins } from '../mixins/PhotoMixins';

/**
 * Default template for MerjentApp - binds mixin methods and defines composite behaviors.
 *
 * This template is automatically installed by the MerjentApp constructor and provides
 * a complete working app configuration. It binds methods from the mixin modules to
 * the app instance and defines composite methods that orchestrate the workflow.
 *
 * ## What This Template Provides
 *
 * - **Two-state status** (Draft/Complete) via {@link StatusMixins.draft_complete}
 * - **Four-tier permissions** (Admin/Editor/User/Viewer) via {@link PermissionMixins.fourTier}
 * - **Field locking** on submission via {@link LockingMixins.standardLockEditing}
 * - **Complete record locking** via {@link LockingMixins.checkCompleteLockStatus}
 * - **Photo validation** via {@link PhotoMixins.standardValidation}
 *
 * ## Mixin Bindings
 *
 * | App Method | Bound To |
 * |------------|----------|
 * | `getStatus` | {@link StatusMixins.draft_complete} |
 * | `getPermissions` | {@link PermissionMixins.fourTier} |
 * | `baseUpdatePermission` | {@link PermissionMixins.standardUpdate} |
 * | `getLockStatus` | {@link LockingMixins.checkLockStatus} |
 * | `getCompleteLockStatus` | {@link LockingMixins.checkCompleteLockStatus} |
 * | `baseLockEditing` | {@link LockingMixins.standardLockEditing} |
 * | `validatePhoto` | {@link PhotoMixins.standardValidation} |
 * | `limitCaption` | {@link PhotoMixins.limitCaption} |
 * | `validateAllPhotos` | {@link PhotoMixins.validateAllPhotos} |
 *
 * ## Composite Methods Defined Here
 *
 * - `start()` - Entry point: runs updatePermission → updateStatus → updateLock
 * - `updatePermission()` - Calls baseUpdatePermission + addUpdatePermission
 * - `lockEditing()` - Calls baseLockEditing + addLockEditing
 * - `updateLock()` - Toggles lock based on getLockStatus()
 * - `updateStatus()` - Updates status if changed
 *
 * ## Extension Hooks
 *
 * Override these methods in your custom template to add behavior:
 * - `addUpdatePermission(event)` - Additional permission logic
 * - `addLockEditing(event)` - Additional locking logic
 * - `reportIDprefix()` - Custom report ID prefix
 *
 * @namespace DefaultTemplate
 * @see {@link StatusMixins} for status calculation methods
 * @see {@link PermissionMixins} for permission methods
 * @see {@link LockingMixins} for field locking methods
 * @see {@link PhotoMixins} for photo validation methods
 */
export function DefaultTemplate(app) {

  // ===== BIND CORE MIXIN METHODS =====

  // Status methods
  app.getStatus = StatusMixins.draft_complete.bind(app);

  // Permission methods
  app.getPermissions = PermissionMixins.fourTier.bind(app);
  app.baseUpdatePermission = PermissionMixins.standardUpdate.bind(app);
  app.standardAdmin = PermissionMixins.standardAdmin.bind(app);
  app.standardEditor = PermissionMixins.standardEditor.bind(app);
  app.standardUser = PermissionMixins.standardUser.bind(app);
  app.standardViewer = PermissionMixins.standardViewer.bind(app);

  // Locking methods
  app.getLockStatus = LockingMixins.checkLockStatus.bind(app);
  app.getCompleteLockStatus = LockingMixins.checkCompleteLockStatus.bind(app);
  app.baseLockEditing = LockingMixins.standardLockEditing.bind(app);

  // Photo methods
  app.validatePhoto = PhotoMixins.standardValidation.bind(app);
  app.limitCaption = PhotoMixins.limitCaption.bind(app);
  app.validateAllPhotos = PhotoMixins.validateAllPhotos.bind(app);


  // ===== COMPOSITE METHODS =====

  /**
   * @description Runs: updatePermission > updateStatus > if lockOnLoad == true then updateLock
   * @param {*} event
   * @returns {void}
   */
  app.start = (event) => {
    app.updatePermission(event);
    app.updateStatus(event);
    app.lockOnLoad ? app.updateLock(event) : null;
    // if (app.restrictNewRecords === true) {
    //   ON('new-record', app.disableNewRecordCreation);
    // }
    ON('new-record', function() {
      if (app.dnAutoName) {
        SETVALUE(app.dnAutoName, USERFULLNAME());
      };
    });
    ON('load-record', function() {
      if (app.restrictNewRecords === true) {
        app.disableNewRecordCreation();
      }
    });
  };

  /**
   * @description baseUpdatePermission() & addUpdatePermission() combined
   * @param {*} event
   * @returns {void}
   */
  app.updatePermission = (event) => {
    app.baseUpdatePermission(event);
    app.addUpdatePermission(event);
  };

  /**
   * @description baseLockEditing() and addLockEditing() combined
   * @param {*} event
   * @returns {void}
   */
  app.lockEditing = (event) => {
    app.baseLockEditing(event);
    app.addLockEditing(event);
  };

  /**
   * @description Turns the field lock on/off depending on value of getLockStatus()
   * @param {*} event
   * @returns {void}
   */
  app.updateLock = (event) => {
    return app.getLockStatus(event) ? app.lockEditing(event) : app.updatePermission(event);
  };

  /**
   * @description Update status
   * @param {*} event
   * @returns {void}
   */
  app.updateStatus = (event) => {
    if (!app.dnSubmit) return;
    
    if (app.getStatus(event) !== STATUS()) {
      SETSTATUS(app.getStatus(event));
      app.updateLock(event);
    }
  };


  // ===== ADDON HOOKS =====

  /**
   * @description Addon update permission restrictions for setting fields readonly.
   *
   * Override this in your app template to add custom permission logic.
   *
   * @param {*} event
   * @returns {void}
   */
  app.addUpdatePermission = (event) => {
    return null;
  };

  /**
   * @description Addon conditions for field lock.
   *
   * Override this in your app template to add custom locking logic.
   *
   * @param {*} event
   * @returns {void}
   */
  app.addLockEditing = (event) => {
    return null;
  };


  // ===== UTILITY METHODS =====

  /**
   * @description Adds coordinator initials when dnReviewed is checked 'yes'
   * @param {*} event
   * @returns {void}
   */
  app.addCoordinatorInitials = (event) => {
    return event.value === 'yes'
      ? SETVALUE(app.dnCoordinatorInitials, M.userInit())
      : SETVALUE(app.dnCoordinatorInitials, null);
  };

  /**
   * @description Enter a callback function that returns a prefix for the report ID
   * @returns {string|null}
   */
  app.reportIDprefix = () => {
    return null;
  };

  /**
   * @description Set report id to: `{report year}{report month}{report day} T {current hours}{current minutes} _ {user initials}`
   * @param {*} event
   * @returns {void}
   */
  app.enterReportID = (event) => {
    const callback = app.reportIDprefix;

    function baseId() {
      const now = new Date();
      const rpDt = VALUE(app.dnDt) || now;
      const vars = M.getReportIDvars(rpDt);
      const nowtime = now.getHours().toString() + now.getMinutes().toString();
      const output = `${vars.yr}${vars.month}${vars.day}_${nowtime}${vars.init}`;
      return output;
    }

    function enterReportID2() {
      if (!callback()) {
        const reportId = baseId();
        SETVALUE(app.dnReportID, reportId);
        return;
      } else {
        const prefix = callback();
        const reportId = prefix + baseId();
        SETVALUE(app.dnReportID, reportId);
        return;
      }
    }

    SETTIMEOUT(enterReportID2, 300);
  };

  /**
   * @description Disable new record creation. Callback for ON('new-record')
   * @param {*} event
   * @returns {void}
   * @example
   * ON('new-record', mApp.disableNewRecordCreation)
   */
  app.disableNewRecordCreation = (event) => {
    if (!app.newRecordCreators.includes(app.getPermissions())) {
      ALERT('You do not have permissions to create a new record in this app.');
      ON('validate-record', function(event) {
        INVALID('You do not have permissions to create a new record in this app. Please contact your supervisor with any questions.');
      });
    }
  };
}
