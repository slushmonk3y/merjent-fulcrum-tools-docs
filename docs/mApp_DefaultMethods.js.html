/** @namespace MerjentApp.DefaultMethods */

/** 
 * @memberof MerjentApp.DefaultMethods
 * @param {MerjentApp} appSettings 
 * @returns {DefaultMethods}
 */
function DefaultMethods(appSettings){
  return {
    /**
     * @description Runs: updatePermission > updateStatus > if lockOnLoad == true then updateLock* 123
     * 
     * @name MerjentApp.DefaultMethods#start 
     * @method
     * @param {*} event
     * @returns {void}
    */
    start:function(event){
      appSettings.updatePermission(event);
      appSettings.updateStatus(event);
      appSettings.lockOnLoad ? appSettings.updateLock(event) : null
    },

    /**
     * @description Enter a callback function that returns a prefix for the report ID
     * 
     * @name MerjentApp.DefaultMethods#reportIDprefix 
     * @method
     * @returns {string}
    */
    reportIDprefix:function(){
      return null
    },

    /**
     * @description Set report id to: `{report year}{report month}{report day} T {current hours}{current minutes} _ {user initials}`
     * 
     * @name MerjentApp.DefaultMethods#enterReportID 
     * @method
     * @param {*} event
     * @returns {void}
    */
    enterReportID:function(event){
      var callback = this.reportIDprefix

      function baseId(){
        const now = new Date()
        const rpDt = VALUE(mApp.dnDt) || now
        const vars = M.getReportIDvars(rpDt)
        const nowtime = now.getHours().toString() + now.getMinutes().toString()
        const output = `${vars.yr}${vars.month}${vars.day}_${nowtime}${vars.init}`
        return output
      }

      function enterReportID2(){
        if(!callback()){
          var reportId = baseId()
          SETVALUE(mApp.dnReportID,reportId)
          return
        } else {
          var prefix = callback()
          var reportId = prefix + baseId()
          SETVALUE(mApp.dnReportID,reportId)
          return
        }
      }

      SETTIMEOUT(enterReportID2,300)
    },
    /**
     * @description Gets the permission level of the USER: Admin | Editor | User | Viewer
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {string}
    */
    getPermissions:function(event){
      return appSettings.adminNames.includes(USERFULLNAME()) || appSettings.adminRoles.includes(ROLE()) ? 'Admin':
        appSettings.editorNames.includes(USERFULLNAME()) || appSettings.editorRoles.includes(ROLE()) ? 'Editor':
          appSettings.userNames.includes(USERFULLNAME()) || appSettings.userRoles.includes(ROLE()) ? 'User':
            appSettings.viewerNames.includes(USERFULLNAME()) || appSettings.viewerRoles.includes(ROLE()) ? 'Viewer':
              appSettings.defaultPermissionLevel;
    },
    /**
     * @description Is the status condition for this.completeLock true
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {boolean}
    */
    getCompleteLockStatus:function(event){
      if(!appSettings.completeLockCondition || !appSettings.getStatus(event)){
        return false; 
      }
      return appSettings.completeLockCondition.includes(appSettings.getStatus(event));
    },
    /**
     * @description Base update permission restrictions for setting fields readonly.
     * 
     * this.updatePermission calls this.baseUpdatePermission() & this.addUpdatePermission() combined.
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
    */
    baseUpdatePermission:function(event){
      if(appSettings.getPermissions()=="Admin"){
        appSettings.getCompleteLockStatus(event) ? M.setAllRO(true,appSettings.completeLockExclude) : M.setAllRO(false)
      } 
      if(appSettings.getPermissions()=="Editor"){
        appSettings.getCompleteLockStatus(event) ? M.setAllRO(true,appSettings.completeLockExclude) : M.setAllRO(null)
        if(appSettings.disableLocationEdit)M.disableLocationEdit()
      } 
      if(appSettings.getPermissions()=="User"){
        appSettings.getCompleteLockStatus(event) ? M.setAllRO(true) : M.setAllRO(null)
        if(appSettings.disableLocationEdit)M.disableLocationEdit()
      } 
      if(appSettings.getPermissions()=="Viewer"){
        M.setAllRO(true)
        M.disableLocationEdit()
      } 
    },
    /**
     * @description Addon update permission restrictions for setting fields readonly.
     * 
     * this.updatePermission calls this.baseUpdatePermission() & this.addUpdatePermission() combined.
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
    */
    addUpdatePermission:function(event){
      return null
    },
    /**
     * @description baseUpdatePermission() & addUpdatePermission() combined
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
    */
    updatePermission:function(event) {
      appSettings.baseUpdatePermission(event)
      appSettings.addUpdatePermission(event)
    },
    /**
     * @description Update status
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {string}
    */
    getStatus:function(event){
      return VALUE(appSettings.dnSubmit) == 'yes' ? 'Complete' : 'Pending'
    },
    /**
     * @description Description placeholder
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
    */
    updateStatus:function(event){
      if(!appSettings.dnSubmit){
        return;
      }
      if(appSettings.getStatus(event) != STATUS()){
        SETSTATUS(appSettings.getStatus(event));
        appSettings.updateLock(event);
      }
    },
    /**
     * @description Is the status condition for this.fieldLock true
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {boolean}
    */
    getLockStatus:function(event){
      if(!appSettings.fieldLockCondition || !appSettings.getStatus(event)){
        return false; 
      }
      return appSettings.fieldLockCondition.includes(appSettings.getStatus(event));
    },
    /**
     * @description Turns the field lock on/off depending on value of getLockStatus(). Defaults to all fields except this.dnSubmit
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
    */
    updateLock:function(event){
      return appSettings.getLockStatus(event) ? appSettings.lockEditing(event) : appSettings.updatePermission(event)
    },
    /**
     * @description Checks for field lock conditions and calls updateLock(). Default condition is if this.dnSubmit == 'yes' 
     * 
     * this.lockEditing calls this.baseLockEditing() & this.addLockEditing() combined.
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
    */
    baseLockEditing:function(event){
      if(appSettings.getPermissions()=="Admin"){
        M.setAllRO(null)
      } 
      if(appSettings.getPermissions()=="Editor"){
        M.setAllRO(null)
      } 
      if(appSettings.getPermissions()=="User"){
        M.setAllRO(true,appSettings.fieldLockExclude)
      } 
      if(appSettings.getPermissions()=="Viewer"){
        M.setAllRO(true)
      } 
    },
    /**
     * @description Addon conditions for field lock. 
     * 
     * this.lockEditing calls this.baseLockEditing() & this.addLockEditing() combined.
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
    */
    addLockEditing:function(event){
      return null
    },
    /**
     * @description baseLockEditing() and  addLockEditing() combined
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
    */
    lockEditing:function(event) {
      appSettings.baseLockEditing();
      appSettings.addLockEditing();
    },
    /**
     * @description Calls photo validation methods if this.requireLandscape = true or this.setLocoff1photo = true;
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
    */
    validatePhoto:function(event){
      appSettings.requireLandscape ? M.photoLandscape(event) : null
      appSettings.setLocoff1photo ? M.photoSetLoc(event) : null
    },
    /**
     * @description Adds coordinators intials when this.dnReviewed is checked 'yes'
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
    */
    addCoordinatorInitials: function(event){
      return event.value=='yes' ? SETVALUE(appSettings.dnCoordinatorInitials,M.getUserInit()):
        SETVALUE(appSettings.dnCoordinatorInitials,null)
    },
    /**
     * @description Alert for text in photo caption exceeding this.photoCapMax length.
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @default 1258
     * @param {*} event
     * @returns {void}
    */
    limitCaption:function(event){
      var max =  appSettings.photoCapMax
      var dn = event.field;
      M.checkPhotoCapLen(dn,max)
    },
    /**
     * @description Alert for record submitted without photo captions
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
    */
    validateAllPhotos:function(event){
      var dn = event.field
      var dns = appSettings.allPhotos
      if(VALUE(dn) == 'yes'){
        dns.forEach(function(dn){
          if(VALUE(dn)){
            M.check4PhotoCaps(dn)
          }
        })
      }
    },
    /**
     * @description Disable new record creation. Callback for ON('new-record'
     * 
     * @memberof MerjentApp.DefaultMethods#
     * @method
     * @param {*} event
     * @returns {void}
     * @example
     * ON('new-record', mApp.disableNewRecordCreation)
    */
    disableNewRecordCreation:function(event){
      if (appSettings.getPermissions() != 'Admin') {
        ALERT('You do not have permissions to create a new record in this app.');
        ON('validate-record', function (event) {
          INVALID('You do not have permissions to create a new record in this app.  Please contact your supervisor with any questions.')
        })
      }
    }
  }
}

export { DefaultMethods };
